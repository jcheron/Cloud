 21             foreach(scandir($dir) as $f) {
 22                 if(!$f || $f[0] == '.') {
 23                     continue;
 24                 }

 26                     $files[] = array(
 27                             "name" => $f,
 28                             "type" => "folder",

 30                             "items" => self::scan($dir . '/' . $f,$rootDir)
 31                     );
 32                 }else {
 33                     $files[] = array(
 34                             "name" => $f,
 35                             "type" => "file",
 36                             "path" => $dir . '/' . $f,
 37                             "size" => filesize($dir . '/' . $f)

 39                 }
